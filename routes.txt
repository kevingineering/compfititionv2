Routes: 
  /goals
    POST    /                   add goal
    GET     /                   all goals
    GET     /:goalId            user goal
    PATCH   /:goalId            update goal
    PATCH   /tracker            update tracker
    DELETE  /:goalId            delete goal

  /user
    POST    /register           register user
    POST    /login              login user
    GET     /                   get user
    GET     /searchable         get searchable users
    PUT     /                   update user
    PATCH   /changepassword     change password
    DELETE  /                   delete user

  /userfriendrequest
    POST    /:friendId          add friend request
    GET     /                   get friend requests
    DELETE  /:friendId          delete friend request
  
  /userfriend
    POST    /:friendId          add friend (and delete friend request)
    GET     /                   get user friends
    GET     /goals/:friendId    get friend's public goals
    GET     /friends/:friendId  get friend's searchable friends
    DELETE  /:friendId          delete friend

  /errorgenerator
    GET     /notfound           test 404 error
    GET     /servererror        test 500 error
    GET     /badrequest         test 400 error
    GET     /badrequest/string  test 400 error with validation

  /competition
    POST    /                   add competition
    GET     /                   get competition
    PATCH   /                   update competition goals
    PATCH   /                   add user to competition
    PATCH   /                   remove user from competition
    PATCH   /                   kick user from competition
    PATCH   /                   add admin to competition
    PATCH   /                   remove admin from competition
    DELETE  /                   delete competition

  /letter
    POST    /                   add new letter
    GET     /:letterId          get letter
    GET     /                   get user letters
    GET     /:compId            get competition letters
    PATCH   /:compId            update competition letter expiration date
    DELETE  /:letterId          delete letter

  /chat
